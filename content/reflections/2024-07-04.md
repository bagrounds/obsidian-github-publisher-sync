function depth(root: TreeNode | null, max = 0): number {  
  if (!root) return max  
  const leftDepth: number = depth(root.left, max + 1)  
  const rightDepth: number = depth(root.right, max + 1)  
  return Math.max(leftDepth, rightDepth)  
}
