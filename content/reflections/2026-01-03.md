// Update 'updated' timestamp in specific index files  
const filesToUpdate = [  
  "index.md",  
  "articles/index.md",  
  "books/index.md",  
  "bot-chats/index.md",  
  "people/index.md",

// ISO 8601 Timestamp (Local Time with Offset)  
// Use .toISOString() instead if you prefer UTC (Zulu) time.  
const timestamp = moment().format();   
  
for (const path of filesToUpdate) {  
  const targetFile = tp.app.vault.getAbstractFileByPath(path);  
    
  // Check if file exists and is actually a file (not a folder/null)  
  if (targetFile instanceof tp.obsidian.TFile) {  
    await tp.app.fileManager.processFrontMatter(targetFile, (frontmatter) => {  
      frontmatter["updated"] = timestamp;  
    });  
  }  
}
